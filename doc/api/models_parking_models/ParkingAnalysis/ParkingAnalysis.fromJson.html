factory ParkingAnalysis.fromJson(Map<String, dynamic> json) {
  List<IllegalCar> illegal = [];
  if (json['illegal_cars'] != null) {
    illegal = (json['illegal_cars'] as List).map((i) => IllegalCar.fromJson(i)).toList();
  }

  List<DoubleParkingViolation> doubleParking = [];
  if (json['double_parking'] != null) {
    doubleParking = (json['double_parking'] as List).map((d) => DoubleParkingViolation.fromJson(d)).toList();
  }

  List<VehicleMatch> matches = [];
  if (json['matches'] != null) {
    matches = (json['matches'] as List).map((m) => VehicleMatch.fromJson(m)).toList();
  }

  List<UnauthorizedVehicle> unauthorized = [];
  if (json['unauthorized_vehicles'] != null) {
    unauthorized = (json['unauthorized_vehicles'] as List).map((u) => UnauthorizedVehicle.fromJson(u)).toList();
  }

  List<NonFacultyViolation> nonFaculty = [];
  if (json['non_faculty_violation_vehicles'] != null) {
    nonFaculty = (json['non_faculty_violation_vehicles'] as List).map((n) => NonFacultyViolation.fromJson(n)).toList();
  }

  int totalSpots = _parseIntSafely(json['total_spots']);
  List<int> freeSpots = _parseIntList(json['free_spots'] ?? []);
  List<int> occupiedSpots = _parseIntList(json['occupied_spots'] ?? []);
  List<int> violationSpots = _parseIntList(json['violation_spots'] ?? []);

  return ParkingAnalysis(
    totalSpots: totalSpots,
    freeSpots: freeSpots,
    occupiedSpots: occupiedSpots,
    violationSpots: violationSpots,
    illegalCars: illegal,
    doubleParkingViolations: doubleParking,
    matches: matches,
    unauthorizedVehicles: unauthorized,
    nonFacultyViolationVehicles: nonFaculty,
  );
}
