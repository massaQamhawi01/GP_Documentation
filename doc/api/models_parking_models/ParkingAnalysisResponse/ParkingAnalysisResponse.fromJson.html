factory ParkingAnalysisResponse.fromJson(Map<String, dynamic> json) {
  final vehicles = <Vehicle>[];
  if (json['vehicles'] != null) {
    for (final v in json['vehicles']) {
      final lpObj   = v['license_plate'];
      final lpText  = lpObj != null ? (lpObj['text'] ?? '') : '';
      if (lpText.isEmpty || lpText.toLowerCase() == 'unknown') {
        continue;                          // ← skip “Unknown” plates
      }
      vehicles.add(
        Vehicle(
          vehicleID: 'VH-${v['id'] ?? DateTime
              .now().millisecondsSinceEpoch}',
          licensePlate: lpText,
          userID: '',
        ),
      );
    }
  }
  return ParkingAnalysisResponse(
    vehicles: vehicles,
    parkingAnalysis: ParkingAnalysis.fromJson(json['parking_analysis'] ?? {}),
    outputImagePath: json['output_image_path'],
  );
}
