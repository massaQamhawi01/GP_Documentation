factory DoubleParkingViolation.fromJson(Map<String, dynamic> json) {
  // pull out the raw values (they might be null)
  final rawCarIdx     = json['car_idx'];
  final rawSpotId     = json['spot_id'];
  final rawNeighborId = json['neighbor_id'];
  final rawOverlap    = json['overlap_ratio'];
  final rawLP         = json['license_plate'];

  // Safe conversion methods for each type
  int safeInt(dynamic value) {
    if (value is int) return value;
    if (value is double) return value.toInt();
    if (value is String) return int.tryParse(value) ?? 0;
    return 0;
  }

  double safeDouble(dynamic value) {
    if (value is double) return value;
    if (value is int) return value.toDouble();
    if (value is String) return double.tryParse(value) ?? 0.0;
    return 0.0;
  }

  return DoubleParkingViolation(
    carIdx: safeInt(rawCarIdx),
    spotId: safeInt(rawSpotId),
    neighborId: safeInt(rawNeighborId),
    overlapRatio: safeDouble(rawOverlap),
    licensePlate: rawLP != null
        ? LicensePlateInfo.fromJson(rawLP as Map<String,dynamic>)
        : LicensePlateInfo(text: '', confidence: 0.0),
  );
}
