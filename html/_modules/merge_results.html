class TwoViewAnalyzer:
    """
    Analyzes and merges vehicle detection results from two camera views.
    
    This class processes pairs of images from different camera angles showing
    the same parking area, merges the detection results, and identifies various
    parking violations by combining information from both views.
    
    The merged results provide a more complete and accurate representation of
    the parking area by eliminating duplicates and resolving conflicts between
    the two views.
    
    Attributes:
        detector (VehicleDetector): Instance of the vehicle detector used for processing images
    """

    def __init__(self):
        """
        Initialize the TwoViewAnalyzer with a VehicleDetector instance.
        """
        self.detector = VehicleDetector()

    def process_pair(
            self,
            img1_path: str,
            img2_path: str,
            parking_data_path: Optional[str] = None,
            regions: Tuple[str, ...] = ("jo",),
    ) -> Dict[str, Any]:
        """
        Process a pair of images from different camera angles and merge the results.
        
        Args:
            img1_path (str): Path to the first camera view image
            img2_path (str): Path to the second camera view image
            parking_data_path (Optional[str], optional): Path to parking spot data JSON file.
                Defaults to None.
            regions (Tuple[str, ...], optional): License plate region codes.
                Defaults to ("jo",) for Jordan.
                
        Returns:
            Dict[str, Any]: Merged detection results containing:
                - Vehicle information
                - Parking analysis including matched spots, violations, etc.
                - Output image paths
        """
        r1 = self.detector.process_image(img1_path, parking_data_path, regions)
        r2 = self.detector.process_image(img2_path, parking_data_path, regions)
        return self._merge_pair_results(r1, r2)



    @staticmethod
    def _spot_map(res: Dict[str, Any]) -> Dict[int, int]:
        """
        Create a map of vehicle indices to their matched parking spot IDs.
        
        Args:
            res (Dict[str, Any]): Detection results from a single camera view
            
        Returns:
            Dict[int, int]: Mapping from vehicle index to parking spot ID
        """
        return {
            m["car_idx"]: m["matched_spot_id"]
            for m in res.get("parking_analysis", {}).get("matches", [])
        }

    @staticmethod
    def _fill_missing_plates(
            violations: List[Dict[str, Any]], vehicles: List[Dict[str, Any]]
    ) -> None:
        """
        Add license plate information to violation entries that are missing it.
        
        Searches through the vehicles list to find matching license plates for
        violation entries that don't have license plate information.
        
        Args:
            violations (List[Dict[str, Any]]): List of violation entries
            vehicles (List[Dict[str, Any]]): List of detected vehicles with their information
        """
        id2plate = {
            v["id"]: v.get("license_plate")
            for v in vehicles
            if v.get("license_plate")
        }

        for v in violations:
            lp = v.get("license_plate")
            if lp and lp is not None:
                continue

            vid = v.get("car_idx")
            if vid in id2plate:
                src = id2plate[vid]
                if isinstance(src, dict):
                    v["license_plate"] = {
                        "plate": src.get("plate") or src.get("text", ""),
                        "score": src.get("score", src.get("confidence", 0.0)),
                    }
                else:
                    v["license_plate"] = src

    @staticmethod
    def _plate(v: Dict[str, Any]) -> str:
        """
        Extract the license plate text from a vehicle dictionary.
        
        Args:
            v (Dict[str, Any]): Vehicle information dictionary
            
        Returns:
            str: License plate text in uppercase, or empty string if not available
        """
        lp = v.get("license_plate") or {}
        return (lp.get("text") or lp.get("plate") or "").upper()

    @staticmethod
    def _is_responsible(camera_id: int, spot_id: Optional[int]) -> bool:
        """
        Check if a camera is responsible for analyzing a specific parking spot.
        
        This helps avoid duplicate analysis by assigning each spot to a specific camera:
        - Camera 1 handles even-numbered spots
        - Camera 2 handles odd-numbered spots
        
        Args:
            camera_id (int): Camera identifier (1 or 2)
            spot_id (Optional[int]): Parking spot ID
            
        Returns:
            bool: True if the camera is responsible for analyzing the spot, False otherwise
        """
        if spot_id is None:
            return True
        return (spot_id % 2 == 0 and camera_id == 1) or (
                spot_id % 2 == 1 and camera_id == 2
        )

    @staticmethod
    def detect_vehicles_missing_permit(
            vehicles: List[Dict[str, Any]],
            allowed_plates: Set[str],
            reason: str, 
    ) -> List[Dict[str, Any]]:
        """
        Detect vehicles whose license plates are not in the allowed permit list.
        
        Args:
            vehicles (List[Dict[str, Any]]): List of detected vehicles
            allowed_plates (Set[str]): Set of allowed license plate numbers
            reason (str): Violation reason (e.g., "Unauthorized vehicle" or "Non faculty")
            
        Returns:
            List[Dict[str, Any]]: List of vehicles without valid permits
        """
        out = []
        for v in vehicles:
            plate_info = v.get("license_plate")
            plate_txt = (plate_info or {}).get("text", "").upper()
            if plate_txt and plate_txt not in allowed_plates:
                out.append(
                    {
                        "car_idx": v["id"],
                        "car_box": v["box"],
                        "license_plate": plate_info,
                        "reason": reason,
                    }
                )
        return out



    @staticmethod
    def _assign_unique_car_ids(
            vehicles: List[Dict[str, Any]], matches: List[Dict[str, Any]]
    ) -> None:
        """
        Assign unique and consistent carIDs to vehicles.
        
        Uses license plates and parking spot information to ensure that
        the same vehicle gets the same ID across different frames and views.
        
        Args:
            vehicles (List[Dict[str, Any]]): List of detected vehicles
            matches (List[Dict[str, Any]]): List of vehicle-to-spot matches
        """
        id_counter = 0
        lp_to_id: Dict[str, str] = {}
        spot_to_id: Dict[int, str] = {}

        idx_to_spot = {m["car_idx"]: m["matched_spot_id"] for m in matches}

        for v in vehicles:
            lp_txt = TwoViewAnalyzer._plate(v)
            spot = idx_to_spot.get(v["id"])

            if lp_txt:
                if lp_txt in lp_to_id:
                    v["carID"] = lp_to_id[lp_txt]
                else:
                    v["carID"] = f"CAR-{id_counter}"
                    lp_to_id[lp_txt] = v["carID"]
                    id_counter += 1
            elif spot is not None:
                if spot in spot_to_id:
                    v["carID"] = spot_to_id[spot]
                else:
                    v["carID"] = f"CAR-{id_counter}"
                    spot_to_id[spot] = v["carID"]
                    id_counter += 1
            else:
                v["carID"] = f"CAR-{id_counter}"
                id_counter += 1

    @staticmethod
    def _deduplicate_violations(violations: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """
        Remove duplicate violation entries.
        
        Prioritizes entries with license plate information and ensures that
        each unique license plate or vehicle ID appears only once in the result.
        
        Args:
            violations (List[Dict[str, Any]]): List of violation entries
            
        Returns:
            List[Dict[str, Any]]: Deduplicated list of violations
        """
        seen_lp = set()
        seen_idx = set()
        result = []

        for v in violations:
            lp = v.get("license_plate", {})
            lp_text = ""

            if lp:
                lp_text = (lp.get("text") or lp.get("plate") or "").upper()

            if lp_text:
                if lp_text not in seen_lp:
                    seen_lp.add(lp_text)
                    seen_idx.add(v.get("car_idx"))
                    result.append(v)
            elif v.get("car_idx") is None:
                result.append(v)

        for v in violations:
            lp = v.get("license_plate", {})
            lp_text = ""
            if lp:
                lp_text = (lp.get("text") or lp.get("plate") or "").upper()

            if not lp_text and v.get("car_idx") is not None:
                if v.get("car_idx") not in seen_idx:
                    seen_idx.add(v.get("car_idx"))
                    result.append(v)

        return result

    def _merge_pair_results(
            self, r1: Dict[str, Any], r2: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        Merge detection results from two camera views into a single comprehensive result.
        
        This method combines vehicle information, parking spot assignments, and violation
        detections from two different camera angles, resolving conflicts and eliminating
        duplicates to produce a unified result.
        
        The merging process includes:
        1. Selecting vehicles based on camera responsibilities
        2. Combining vehicle information using the best data from each view
        3. Merging and deduplicating various types of violations
        4. Handling conflicts between different violation types
        5. Computing final statistics about parking spots
        
        Args:
            r1 (Dict[str, Any]): Detection results from the first camera
            r2 (Dict[str, Any]): Detection results from the second camera
            
        Returns:
            Dict[str, Any]: Merged detection results containing:
                - Combined vehicle information
                - Comprehensive parking analysis
                - Paths to the output images
        """
        cam1, cam2 = r1["camera_id"], r2["camera_id"]
        s1, s2 = self._spot_map(r1), self._spot_map(r2)

        vehicles_map: Dict[str, Dict[str, Any]] = {}

        def accept(v: Dict[str, Any], cam: int, spot_map: Dict[int, int]) -> bool:
            return self._is_responsible(cam, spot_map.get(v["id"]))

        def vkey(v: Dict[str, Any], spot_map: Dict[int, int]) -> str:
            """
            Generate a unique key for a vehicle based on license plate, spot, or box.
            
            Returns:
                str: Unique identifier for the vehicle
            """
            lp = self._plate(v)
            if lp:
                return f"LP:{lp}"
            spot = spot_map.get(v["id"])
            if spot is not None:
                return f"SPOT:{spot}"
            b = v.get("box") or ()
            return f"BOX:{tuple(round(x / 10) * 10 for x in b)}"

        def _better_plate(old: Optional[dict], new: Optional[dict]) -> Optional[dict]:
            """
            Determine which license plate information is better.
            
            Compares confidence scores to select the better plate.
            
            Returns:
                Optional[dict]: The better license plate information or None
            """
            if not new:
                return old
            if not old:
                return new
            score_old = old.get("confidence", old.get("score", 0))
            score_new = new.get("confidence", new.get("score", 0))
            return new if score_new > score_old else old

        # Merge vehicle information from both cameras
        for res, cam, smap in ((r1, cam1, s1), (r2, cam2, s2)):
            for v in res["vehicles"]:
                if not accept(v, cam, smap):
                    continue
                key = vkey(v, smap)

                if key not in vehicles_map:
                    vehicles_map[key] = dict(v)
                else:
                    tgt = vehicles_map[key]

                    tgt["license_plate"] = _better_plate(
                        tgt.get("license_plate"), v.get("license_plate")
                    )

                    if v.get("confidence", 0) > tgt.get("confidence", 0):
                        tgt["confidence"] = v["confidence"]
                        tgt["box"] = v["box"]

                    if v.get("distance", 1e9) < tgt.get("distance", 1e9):
                        tgt["distance"] = v["distance"]

        vehicles: List[Dict[str, Any]] = list(vehicles_map.values())

        def keep(seq, cam, field):
            """
            Filter a sequence to keep only items that the camera is responsible for.
            
            Returns:
                list: Filtered sequence
            """
            return [x for x in seq if self._is_responsible(cam, x.get(field))]

        pa1, pa2 = r1.get("parking_analysis", {}), r2.get("parking_analysis", {})

        # Merge double parking violations
        doubles = (pa1.get("double_parking", []) or []) + (pa2.get("double_parking", []) or [])
        doubles = self._deduplicate_violations(doubles)

        double_parked_plates = set()
        double_parked_spots = set()

        for dp in doubles:
            lp = dp.get("license_plate") or {}
            plate_txt = (lp.get("text") or lp.get("plate") or "").upper()
            if plate_txt:
                double_parked_plates.add(plate_txt)

            if "spot_id" in dp:
                double_parked_spots.add(dp["spot_id"])
            if "neighbor_id" in dp:
                double_parked_spots.add(dp["neighbor_id"])

        # Merge vehicle-to-spot matches and filter out double-parked vehicles
        all_matches = keep(pa1.get("matches", []), cam1, "matched_spot_id") + keep(pa2.get("matches", []), cam2,
                                                                                   "matched_spot_id")
        filtered_matches = []

        for match in all_matches:
            spot_id = match.get("matched_spot_id")
            if spot_id in double_parked_spots:
                continue

            lp = match.get("license_plate") or {}
            plate_txt = (lp.get("text") or lp.get("plate") or "").upper()
            if plate_txt and plate_txt in double_parked_plates:
                continue

            filtered_matches.append(match)

        matches = sorted(
            filtered_matches,
            key=lambda m: (m.get("license_plate") or {}).get("score", 0),
            reverse=True,
        )

        # Ensure each spot is only assigned once
        seen_spots: Set[int] = set()
        matches = [m for m in matches if
                   not (m["matched_spot_id"] in seen_spots or seen_spots.add(m["matched_spot_id"]))]

        # Merge illegal parking violations
        illegal = keep(pa1.get("illegal_cars", []), cam1, "spot_id") + keep(pa2.get("illegal_cars", []), cam2,
                                                                            "spot_id")
        illegal = self._deduplicate_violations(illegal)

        # Remap vehicle indices to the merged list
        old_ids = [v["id"] for v in vehicles]
        lp_list = [self._plate(v) for v in vehicles]
        old2new = {(old_ids[i], lp_list[i]): i for i in range(len(vehicles))}
        for i, v in enumerate(vehicles):
            v["id"] = i

        def re_map(arr):
            """
            Remap vehicle indices in violation entries to the merged vehicle list.
            
            Also updates license plate information if missing.
            """
            for item in arr:
                car_idx = item.get("car_idx")
                if car_idx is None:
                    continue

                lp = item.get("license_plate") or {}
                txt = (lp.get("plate") or lp.get("text") or "").upper()

                new_idx = None

                if txt:
                    key = (car_idx, txt)
                    if key in old2new:
                        new_idx = old2new[key]

                if new_idx is not None:
                    item["car_idx"] = new_idx
                    # keep data consistent
                    item["car_box"] = vehicles[new_idx]["box"]
                    if not txt:
                        item["license_plate"] = vehicles[new_idx].get("license_plate")

                if new_idx is not None:
                    item["car_idx"] = new_idx

                    if item.get("license_plate") is None or not txt:
                        for v in vehicles:
                            if v["id"] == new_idx:
                                vehicle_lp = v.get("license_plate")
                                if vehicle_lp:
                                    if isinstance(vehicle_lp, dict):
                                        item["license_plate"] = {
                                            "plate": vehicle_lp.get("text", ""),
                                            "score": vehicle_lp.get("confidence", 0.0)
                                        }
                                    else:
                                        item["license_plate"] = vehicle_lp
                                break

        for arr in (matches, illegal, doubles):
            re_map(arr)

        self._assign_unique_car_ids(vehicles, matches)

        spot_map = {m["car_idx"]: m["matched_spot_id"] for m in matches}

        # Detect authorization violations
        try:
            if hasattr(self.detector, 'detect_firestore_violations') and self.detector.firestore_enabled:

                logger.info("Using Firestore for violation detection.")
                unauthorized, non_faculty = self.detector.detect_firestore_violations(vehicles, spot_map)

                unauthorized = self._deduplicate_violations(unauthorized)
                non_faculty = self._deduplicate_violations(non_faculty)
            else:
                raise AttributeError("Firestore not available")
        except Exception as e:
            logger.warning(f"Firestore violation detection failed: {e}\nFalling back to file-based approach")

            def _load(path: str) -> Set[str]:
                """
                Load a set of license plates from a text file.
                
                Returns:
                    Set[str]: Set of license plate numbers in uppercase
                """
                if not os.path.exists(path):
                    return set()
                with open(path) as f:
                    return {ln.strip().upper() for ln in f if ln.strip()}

            general_permits = _load("authorized_plates.txt")
            faculty_permits = _load("faculty_plates.txt")
            faculty_only = {1, 2, 3, 4}

            unauthorized = self.detect_vehicles_missing_permit(
                vehicles, general_permits, "Unauthorized vehicle"
            )
            unauthorized = self._deduplicate_violations(unauthorized)

            vehicles_in_faculty = [
                v for v in vehicles if spot_map.get(v["id"]) in faculty_only
            ]
            non_faculty = self.detect_vehicles_missing_permit(
                vehicles_in_faculty,
                faculty_permits,
                "Non faculty",
            )
            non_faculty = self._deduplicate_violations(non_faculty)

        # Fill in missing license plate information
        self._fill_missing_plates(doubles, vehicles)
        self._fill_missing_plates(illegal, vehicles)
        self._fill_missing_plates(unauthorized, vehicles)
        self._fill_missing_plates(non_faculty, vehicles)

        # Resolve conflicts between different violation types
        dp_ids = {d["car_idx"] for d in doubles}
        dp_plates = {self._plate(d) for d in doubles}

        def _same_vehicle(item):
            """
            Check if an item refers to the same vehicle as one in the doubles list.
            
            Returns:
                bool: True if it's the same vehicle, False otherwise
            """
            if item.get("car_idx") in dp_ids:
                return True
            return self._plate(item) in dp_plates

        match_ids = {m["car_idx"] for m in matches}
        match_plates = {self._plate(m) for m in matches}

        def _in_matches(item):
            """
            Check if an item refers to a vehicle that's in the matches list.
            
            Returns:
                bool: True if it's in matches, False otherwise
            """
            if item.get("car_idx") in match_ids:
                return True
            return self._plate(item) in match_plates

        # Remove illegal parking violations if the vehicle is properly matched
        illegal = [ic for ic in illegal if not _in_matches(ic)]

        # Remove illegal parking violations if the vehicle is already marked as double-parked
        illegal = [ic for ic in illegal if not _same_vehicle(ic)]

        # Compute final spot statistics
        occ_from_matches = {m["matched_spot_id"] for m in matches}
        occ_from_double = {d["spot_id"] for d in doubles} | \
                          {d["neighbor_id"] for d in doubles}

        raw_violations = (
                occ_from_double
                | {i["spot_id"] for i in illegal if "spot_id" in i}
                | {spot_map.get(v["car_idx"]) for v in unauthorized if "car_idx" in v}
                | {spot_map.get(v["car_idx"]) for v in non_faculty if "car_idx" in v}
        )

        violation_spots = sorted(s for s in raw_violations if s is not None)

        occupied_spots = sorted(occ_from_matches | occ_from_double)
        total_spots = pa1.get("total_spots") or pa2.get("total_spots") or 0
        free_spots = sorted(set(range(1, total_spots + 1)) - set(occupied_spots))

        # Create final parking analysis dictionary
        parking_analysis = {
            "total_spots": total_spots,
            "occupied_spots": occupied_spots,
            "free_spots": free_spots,
            "matches": matches,
            "illegal_cars": illegal,
            "double_parking": doubles,
            "unauthorized_vehicles": unauthorized,
            "non_faculty_violation_vehicles": non_faculty,
            "violation_spots": violation_spots,
        }

        return {
            "camera_id": 0,
            "vehicles": vehicles,
            "parking_analysis": parking_analysis,
            "image_pair": (r1.get("output_image_path"), r2.get("output_image_path")),
            "output_image_path": r1.get("output_image_path") or "",
        }
