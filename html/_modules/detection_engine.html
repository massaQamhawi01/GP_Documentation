class FirestoreManager:
    """
    Manages interactions with Firestore database for vehicle detection system.
    
    This class handles all database operations including querying vehicle information,
    checking for existing violations, and retrieving user data associated with 
    license plates.
    
    Attributes:
        db (firestore.Client): Firestore database client instance
        
    Raises:
        Exception: If Firestore connection cannot be established
    """
    
    def __init__(self, service_account_key_path=None):
        """
        Initialize the Firestore manager with service account credentials.
        
        Args:
            service_account_key_path (str, optional): Path to the service account key file.
                Defaults to None, in which case it's read from environment variable.
                
        Raises:
            Exception: If Firestore connection cannot be established
        """
        if service_account_key_path is None:
            service_account_key_path = os.getenv('FIRESTORE_SERVICE_ACCOUNT_KEY', 'service-account-key.json')

        try:
            if not firebase_admin._apps:
                cred = credentials.Certificate(service_account_key_path)
                firebase_admin.initialize_app(cred)

            self.db = firestore.client()
            logger.info("Firestore connection established")
        except Exception as e:
            logger.warning(f"Error connecting to Firestore: {e}")
            raise

    def violation_exists(self, license_plate: str) -> bool:
        """
        Check if a violation record exists for the given license plate.
        
        Args:
            license_plate (str): The license plate to check
            
        Returns:
            bool: True if a violation already exists, False otherwise
            
        Raises:
            Exception: If there's an error querying Firestore
        """
        try:
            qs = (
                self.db.collection('violation')
                .where('licensePlate', '==', license_plate)
                .limit(1)
                .get()
            )
            return len(qs) > 0
        except Exception as e:
            logging.error(f"Error checking existing violation: {e}")
            return True

    def get_user_by_license_plate(self, license_plate: str) -> Optional[Dict[str, Any]]:
        """
        Retrieve user information associated with a license plate.
        
        Args:
            license_plate (str): The license plate to query
            
        Returns:
            Optional[Dict[str, Any]]: User information dictionary if found,
                including user ID, role, name, email and vehicle data.
                Returns None if no matching user is found.
                
        Raises:
            Exception: If there's an error querying Firestore
        """
        try:
            vehicle_ref = self.db.collection('vehicle')
            query = vehicle_ref.where('licensePlate', '==', license_plate).limit(1)

            vehicle_docs = query.get()

            if not vehicle_docs:
                return None

            vehicle_data = vehicle_docs[0].to_dict()
            user_id = vehicle_data.get('userID')

            if not user_id:
                return None

            user_ref = self.db.collection('user').document(user_id)
            user_doc = user_ref.get()

            if user_doc.exists:
                user_data = user_doc.to_dict()
                return {
                    'userID': user_id,
                    'role': user_data.get('role'),
                    'name': user_data.get('name'),
                    'email': user_data.get('email'),
                    'vehicle_data': vehicle_data
                }
            else:
                return None

        except Exception as e:
            logger.error(f"Error querying Firestore: {e}")
            return None

    """
    Main class for vehicle detection, license plate recognition, and parking analysis.
    
    This class handles vehicle detection using YOLO models, license plate recognition
    using PlateRecognizer API, matching vehicles to parking spots, and detecting
    various parking violations.
    
    Attributes:
        VEHICLE_CLASSES (Dict[int, str]): Mapping of YOLO class IDs to vehicle types
        VEHICLE_COLOR (Tuple): RGB color for vehicle bounding boxes
        LP_COLOR (Tuple): RGB color for license plate information
        LEGAL_SPOT_COLOR (Tuple): RGB color for legal parking spots
        ILLEGAL_SPOT_COLOR (Tuple): RGB color for illegal parking violations
        DOUBLE_PARKING_COLOR (Tuple): RGB color for double parking violations
        UNAUTHORIZED_COLOR (Tuple): RGB color for unauthorized vehicle violations
        NON_FACULTY_COLOR (Tuple): RGB color for non-faculty in faculty spot violations
        API_KEY (str): API key for PlateRecognizer service
        MAX_UPLOAD (int): Maximum byte size for image uploads to PlateRecognizer API
        MAX_RPS (int): Maximum requests per second allowed by PlateRecognizer API
        KNOWN_WIDTH (float): Estimated real-world width of a typical car in meters
        FOCAL_LENGTH (int): Focal length used for estimating vehicle distance
        firestore_manager (FirestoreManager): Manager for Firestore database operations
        firestore_enabled (bool): Whether Firestore functionality is available
        device (str): Device used for model inference ('cuda' or 'cpu')
        model (YOLO): YOLO model instance for vehicle detection

    def _draw_label(img, txt: str, origin: Tuple[int, int], bg: Tuple[int, int, int]):
        """
        Draw a text label with background on the image.
        
        Args:
            img (numpy.ndarray): Image to draw on
            txt (str): Text to display in the label
            origin (Tuple[int, int]): Bottom-left coordinate of the text
            bg (Tuple[int, int, int]): Background color of the label (BGR format)

        x, y = origin
        cv2.rectangle(img, (x, y - h - bl - 4), (x + w, y), bg, -1)
        fg = (255, 255, 255) if sum(bg) < 200 else (0, 0, 0)
        cv2.putText(img, txt, (x, y - 2), cv2.FONT_HERSHEY_SIMPLEX, 0.5, fg, 2)

    def _enhance(self, crop: np.ndarray, upscale: int = 2) -> np.ndarray:
        """
        Enhance an image crop for better license plate recognition.
        
        Applies upscaling, contrast enhancement, and sharpening to improve
        license plate visibility for OCR.
        
        Args:
            crop (numpy.ndarray): Image crop to enhance
            upscale (int, optional): Upscaling factor. Defaults to 2.
            
        Returns:
            numpy.ndarray: Enhanced image
        """
        if upscale != 1:
            crop = cv2.resize(crop, None, fx=upscale, fy=upscale, interpolation=cv2.INTER_CUBIC)
        lab = cv2.cvtColor(crop, cv2.COLOR_BGR2LAB)
        l, a, b = cv2.split(lab)
        l = cv2.createCLAHE(3.0, (8, 8)).apply(l)
        crop = cv2.cvtColor(cv2.merge((l, a, b)), cv2.COLOR_LAB2BGR)
        blur = cv2.GaussianBlur(crop, (0, 0), 3)
        return cv2.addWeighted(crop, 1.6, blur, -0.6, 0)

    def _wait_for_slot(self):
        """
        Implement rate limiting for API requests.
        
        Waits if necessary to ensure API rate limits are not exceeded.
        Updates the request times list and removes old entries.
        """
        now = time.time()
        self._req_times = [t for t in self._req_times if now - t < 1.0]
        if len(self._req_times) >= self.MAX_RPS:
            to_sleep = 0.5 - (now - self._req_times[0]) + 0.05
            if to_sleep > 0:
                time.sleep(to_sleep)

            total_spots (int, optional): Total number of spots. Defaults to 14.

    def _post_to_lpr(self, img_bgr: np.ndarray, regions=("jo",)) -> Optional[Dict[str, Any]]:
        """
        Send license plate image to Plate Recognizer API.
        
        Compresses the image if needed and handles rate limiting and retries.
        
        Args:
            img_bgr (numpy.ndarray): Image containing license plate
            regions (tuple, optional): Region codes for license plate format.
                Defaults to ("jo",) for Jordan.
                
        Returns:
            Optional[Dict[str, Any]]: API response with plate recognition results,
                or None if recognition failed or API call failed
                
        Raises:
            RuntimeError: If JPEG encoding fails
        """
        self._wait_for_slot()

        tmp = img_bgr.copy()
        q = 90
        while True:
            ok, buf = cv2.imencode(".jpg", tmp, [int(cv2.IMWRITE_JPEG_QUALITY), q])
            if not ok:
                raise RuntimeError("JPEG encode failed")
            if len(buf) <= self.MAX_UPLOAD or (q <= 30 and min(tmp.shape[:2]) < 300):
                break
            q = 80 if q > 40 else q - 10
            if q <= 40:
                tmp = cv2.resize(tmp, None, fx=0.7, fy=0.7, interpolation=cv2.INTER_AREA)
                q = 80

        for attempt in range(5):
            response = requests.post(
                "https://api.platerecognizer.com/v1/plate-reader/",
                data=dict(regions=regions),
                files=dict(upload=("crop.jpg", buf.tobytes(), "image/jpeg")),
                headers={"Authorization": f"Token {self.API_KEY}"},
                timeout=20,
            )
            self._req_times.append(time.time())

            if response.status_code == 201:
                data = response.json()
                return data if data.get("results") else None
            if response.status_code == 429:
                retry_after = float(response.headers.get("Retry-After", 1))
                jitter = random.uniform(0.05, 0.2)
                time.sleep(retry_after + jitter)
                continue
            logger.warning(f"[ALPR] HTTP {response.status_code}: {response.text}")
            return None
        logger.warning("[ALPR] Too many retries, giving up")
        return None

    def _estimate_distance(self, pix_w: int) -> float:
        """
        Estimate the distance to a vehicle based on its pixel width.
        
        Uses the formula: distance = (known_width * focal_length) / pixel_width
        
        Args:
            pix_w (int): Width of the vehicle in pixels
            
        Returns:
            float: Estimated distance in meters
        """
        return (self.KNOWN_WIDTH * self.FOCAL_LENGTH) / pix_w if pix_w else 0.0

    def is_point_right_of_line(self, p, a, b):
        """
        Determine if a point is to the right of a line.
        
        Args:
            p (tuple): Point coordinates (px, py)
            a (tuple): First line endpoint (ax, ay)
            b (tuple): Second line endpoint (bx, by)
            
        Returns:
            bool: True if the point is to the right of the line, False otherwise
        """
        ax, ay = a
        bx, by = b
        px, py = p
        cross = (bx - ax) * (py - ay) - (by - ay) * (px - ax)
        return cross < 0

    def calculate_iou(self, box1, box2):
        """
        Calculate the Intersection over Union (IoU) of two bounding boxes.
        
        Args:
            box1 (list): First bounding box coordinates [x1, y1, x2, y2]
            box2 (list): Second bounding box coordinates [x1, y1, x2, y2]
            
        Returns:
            float: IoU value between 0 and 1
        """
        box1_x1, box1_y1, box1_x2, box1_y2 = box1
        box2_x1, box2_y1, box2_x2, box2_y2 = box2

        x_left = max(box1_x1, box2_x1)
        y_top = max(box1_y1, box2_y1)
        x_right = min(box1_x2, box2_x2)
        y_bottom = min(box1_y2, box2_y2)

        if x_right < x_left or y_bottom < y_top:
            intersection = 0
        else:
            intersection = (x_right - x_left) * (y_bottom - y_top)

        box1_area = (box1_x2 - box1_x1) * (box1_y2 - box1_y1)
        box2_area = (box2_x2 - box2_x1) * (box2_y2 - box2_y1)
        union = box1_area + box2_area - intersection

        iou = intersection / union if union > 0 else 0
        return iou

    def detect_vehicles(self, image):
        """
        Detect vehicles in an image using YOLOv8 model.
        
        Runs the YOLO model with two different configurations to maximize
        detection accuracy, then merges the results while avoiding duplicates.
        
        Args:
            image (numpy.ndarray): Input image
            
        Returns:
            tuple: Three lists containing:
                - Bounding boxes of detected vehicles [x1, y1, x2, y2]
                - Confidence scores
                - Class IDs of detected vehicles
        """
        results1 = self.model(image,
                              device=self.device,
                              conf=0.08,
                              iou=0.35,
                              agnostic_nms=True,
                              max_det=100)

        results2 = self.model(image,
                              device=self.device,
                              conf=0.1,
                              iou=0.25,
                              agnostic_nms=False,
                              max_det=100)

        all_boxes = []
        all_confs = []
        all_cls = []

        for box in results1[0].boxes:
            x1, y1, x2, y2 = box.xyxy[0]
            conf = float(box.conf)
            cls = int(box.cls)

            if cls in self.VEHICLE_CLASSES and conf > 0.08:
                all_boxes.append([x1, y1, x2, y2])
                all_confs.append(conf)
                all_cls.append(cls)

        for box in results2[0].boxes:
            x1, y1, x2, y2 = box.xyxy[0]
            conf = float(box.conf)
            cls = int(box.cls)

            if cls in self.VEHICLE_CLASSES and conf > 0.08:
                is_new = True
                for existing_box in all_boxes:
                    if self.calculate_iou(existing_box, [x1, y1, x2, y2]) > 0.5:
                        is_new = False
                        break

                if is_new:
                    all_boxes.append([x1, y1, x2, y2])
                    all_confs.append(conf)
                    all_cls.append(cls)

        return all_boxes, all_confs, all_cls

    def check_user_violation(self, license_plate: str, spot_id: Optional[int]) -> Optional[str]:
        """
        Check if a vehicle is violating parking rules based on user role.
        
        Args:
            license_plate (str): License plate of the vehicle
            spot_id (Optional[int]): Parking spot ID where the vehicle is parked
            
        Returns:
            Optional[str]: Violation type if a violation is detected, None otherwise
        """
        if not license_plate:
            return "No license plate detected"

        if not self.firestore_enabled:
            return None

        try:
            user_info = self.firestore_manager.get_user_by_license_plate(license_plate)

            if not user_info:
                return "Unauthorized vehicle"

            faculty_only_spots = {1, 2, 3, 4}

            if spot_id and spot_id in faculty_only_spots:
                if user_info.get('role') != 'faculty':
                    return "Non faculty"

            allowed_roles = {'student', 'faculty', 'staff'}
            if user_info.get('role') not in allowed_roles:
                return "Unauthorized vehicle"

            return None
        except Exception as e:
            logger.error(f"Error during Firestore violation check: {e}")
            return None

    def detect_firestore_violations(self, vehicles: List[Dict[str, Any]], spot_map: Dict[int, int]) -> Tuple[
        List[Dict], List[Dict]]:
        """
        Detect parking violations based on user roles and parking spots.
        
        Args:
            vehicles (List[Dict[str, Any]]): List of detected vehicles with their data
            spot_map (Dict[int, int]): Mapping of vehicle IDs to parking spot IDs
            
        Returns:
            Tuple[List[Dict], List[Dict]]: Two lists containing:
                - Unauthorized vehicles
                - Non-faculty vehicles parked in faculty spots
                
        Raises:
            Exception: If there's an error checking violations in Firestore
        """
        unauthorized = []
        non_faculty = []

        for vehicle in vehicles:
            license_plate = self._plate(vehicle)
            spot_id = spot_map.get(vehicle["id"])

            violation_type = self.check_user_violation(license_plate, spot_id)

            if violation_type:
                violation_data = {
                    "car_idx": vehicle["id"],
                    "car_box": vehicle["box"],
                    "license_plate": vehicle.get("license_plate"),
                    "reason": violation_type
                }

                if "Unauthorized" in violation_type:
                    unauthorized.append(violation_data)
                elif "Non faculty" in violation_type:
                    non_faculty.append(violation_data)

        return unauthorized, non_faculty

    def process_image(self, image_path: str, parking_data_path: Optional[str] = None, regions=("jo",)):
        """
        Process an image for vehicle detection and parking analysis.
        
        This is the main entry point for the detection pipeline, which:
        1. Detects vehicles in the image
        2. Recognizes license plates
        3. Performs parking spot analysis if parking data is provided
        4. Detects parking violations
        5. Creates an annotated output image
        
        Args:
            image_path (str): Path to the input image
            parking_data_path (Optional[str], optional): Path to parking spot data JSON file.
                Defaults to None.
            regions (tuple, optional): License plate region codes. Defaults to ("jo",).
            
        Returns:
            Dict: Results dictionary containing:
                - Detected vehicles with their information
                - Parking analysis results (if applicable)
                - Path to the output image
                
        Raises:
            FileNotFoundError: If the input image cannot be read
        """
        frame = cv2.imread(image_path)
        if frame is None:
            raise FileNotFoundError(f"Could not read image at {image_path}")

        cam_num = None
        parent = os.path.basename(os.path.dirname(image_path)).lower()

        if parent == 'images1':
            cam_num = 1
        elif parent == 'images2':
            cam_num = 2
        elif parent == 'images3':
            cam_num = 3

        all_boxes, all_confs, all_cls = self.detect_vehicles(frame)
        logger.info(f"Detected {len(all_boxes)} vehicles")

        output_image = frame.copy()
        h, w = frame.shape[:2]

        license_plates = {}
        vehicle_data = []

        for i in range(len(all_boxes)):
            x1, y1, x2, y2 = map(int, all_boxes[i])

            conf = all_confs[i]
            cls = all_cls[i]
            vehicle_type = self.VEHICLE_CLASSES.get(cls, "Unknown")

            pixel_width = x2 - x1
            distance = self._estimate_distance(pixel_width)

            padx = int(0.05 * (x2 - x1))
            height = y2 - y1
            lp_y1 = y2 - int(0.35 * height)
            lp_y2 = y2 + int(0.05 * height)
            cx1 = max(0, x1 - padx)
            cx2 = min(w, x2 + padx)
            cy1 = max(0, lp_y1)
            cy2 = min(h, lp_y2)

            crop = frame[cy1:cy2, cx1:cx2]
            enhanced_crop = self._enhance(crop)

            alpr = self._post_to_lpr(enhanced_crop, regions) if self.API_KEY else None
            plate_txt, plate_score = None, 0.0

            if alpr and alpr.get("results"):
                best_res = max(alpr["results"], key=lambda r: r["score"])
                plate_txt = best_res["plate"].upper()
                plate_score = best_res["score"]

                if "candidates" in best_res:
                    for cand in best_res["candidates"]:
                        cand_plate = cand["plate"].upper()
                        cand_score = cand["score"]
                        if len(cand_plate) > len(plate_txt) and cand_score >= plate_score - 0.05:
                            plate_txt, plate_score = cand_plate, cand_score

                license_plates[i] = {
                    "plate": plate_txt,
                    "score": plate_score
                }

            cv2.rectangle(output_image, (x1, y1), (x2, y2), self.VEHICLE_COLOR, 2)

            self._draw_label(output_image, f"{vehicle_type} ({conf:.2f})", (x1, y1), self.VEHICLE_COLOR)

            cv2.putText(output_image, f"{distance:.2f} m", (x1, y2 + 20),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.VEHICLE_COLOR, 2)

            if plate_txt:
                self._draw_label(output_image, f"LP: {plate_txt} ({plate_score:.2f})",
                                 (x1, y1 - 20), self.LP_COLOR)

            vehicle_info = {
                "id": i,
                "type": vehicle_type,
                "confidence": float(conf),
                "box": [int(x1), int(y1), int(x2), int(y2)],
                "distance": float(distance),
                "license_plate": {
                    "text": plate_txt,
                    "confidence": float(plate_score) if plate_txt else 0.0
                } if plate_txt else None
            }
            vehicle_data.append(vehicle_info)

        results = {
            "vehicles": vehicle_data,
            "image_width": w,
            "image_height": h,
            "camera_id": cam_num,
            "parking_analysis": None,
            "output_image_path": None
        }

        parking_results = None

        if parking_data_path and cam_num:
            output_image, parking_results = self.analyze_parking(output_image, all_boxes, parking_data_path, cam_num)

            if parking_results:
                for match in parking_results.get('matches', []):
                    car_idx = match['car_idx']
                    match['license_plate'] = license_plates.get(car_idx, None)

                for illegal in parking_results.get('illegal_cars', []):
                    car_idx = illegal['car_idx']
                    illegal['license_plate'] = license_plates.get(car_idx, None)

                for dp in parking_results.get('double_parking', []):
                    car_idx = dp['car_idx']
                    dp['license_plate'] = license_plates.get(car_idx, None)

                spot_map = {m['car_idx']: m['matched_spot_id'] for m in parking_results.get('matches', [])}

                if self.firestore_enabled:
                    try:
                        unauthorized, non_faculty = self.detect_firestore_violations(vehicle_data, spot_map)

                        all_new = (
                                parking_results.get('illegal_cars', []) +
                                parking_results.get('double_parking', []) +
                                unauthorized +
                                non_faculty
                        )
                        self._save_new_violations(all_new, violation_type_field="reason")

                        parking_results['unauthorized_vehicles'] = unauthorized
                        parking_results['non_faculty_violation_vehicles'] = non_faculty

                        self._draw_firestore_violations(
                            output_image,
                            unauthorized,
                            non_faculty,
                            parking_results.get('illegal_cars', []),
                            parking_results.get('double_parking', [])
                        )
                    except Exception as e:
                        logger.error(f"Error detecting Firestore violations: {e}")
                        parking_results['unauthorized_vehicles'] = []
                        parking_results['non_faculty_violation_vehicles'] = []
                else:
                    parking_results['unauthorized_vehicles'] = []
                    parking_results['non_faculty_violation_vehicles'] = []

                results["parking_analysis"] = parking_results

        output_path = f'output_detection_cam{cam_num}.jpg' if cam_num else 'output_detection.jpg'
        cv2.imwrite(output_path, output_image)
        results["output_image_path"] = output_path

        return results

    def load_parking_spots(self, filename: str, camera_id: int):
        """
        Load parking spot data from a JSON file.
        
        Args:
            filename (str): Path to the JSON file containing parking spot data
            camera_id (int): Camera ID to filter spots for the specific camera view
            
        Returns:
            List[Dict]: List of parking spots with their polygons and metadata
        """
        with open(filename, 'r') as f:
            data = json.load(f)

        parking_spots = []

        for ann in data['annotations']:
            if ann['image_id'] != camera_id:
                continue

            category_name = None
            for cat in data['categories']:
                if cat['id'] == ann['category_id']:
                    category_name = cat['name']
                    break

            if ann['segmentation'] and len(ann['segmentation']) > 0:
                polygon_coords = np.array(ann['segmentation'][0]).reshape(-1, 2)
                polygon = Polygon(polygon_coords)

                x_center = np.mean(polygon_coords[:, 0])
                bottom_y = max(polygon_coords[:, 1])

                parking_spots.append({
                    'polygon': polygon,
                    'id': ann['category_id'],
                    'name': category_name,
                    'x_center': x_center,
                    'bottom_y': bottom_y,
                    'coords': polygon_coords
                })

        parking_spots.sort(key=lambda x: x['id'])
        return parking_spots

    def match_cars_to_parking_spots(self, car_boxes, parking_spots, column_threshold):
        """
        Match detected vehicles to parking spots.
        
        Args:
            car_boxes (List): List of vehicle bounding boxes
            parking_spots (List[Dict]): List of parking spots
            column_threshold (int): Maximum horizontal distance for matching
            
        Returns:
            Tuple[List[Dict], List[int]]: A tuple containing:
                - List of matches (car to spot)
                - List of free (unoccupied) spot IDs
        """
        matches = []
        matched_car_indices = set()
        occupied_spots = set()

        for car_idx, car_box in enumerate(car_boxes):
            car_x_mid, car_bottom_y = self.get_car_mid_bottom(car_box)
            car_point = Point(car_x_mid, car_bottom_y)

            candidates = []
            for spot in parking_spots:
                if spot['id'] in occupied_spots:
                    continue

                min_x, min_y, max_x, max_y = spot['polygon'].bounds
                margin = column_threshold
                if (min_x - margin) <= car_x_mid <= (max_x + margin):
                    distance = spot['polygon'].distance(car_point)
                    candidates.append((distance, spot))

            if candidates:
                _, best_spot = min(candidates, key=lambda x: x[0])
                matches.append({
                    'car_idx': car_idx,
                    'car_box': car_box,
                    'matched_spot_id': best_spot['id']
                })
                matched_car_indices.add(car_idx)
                occupied_spots.add(best_spot['id'])
            else:
                logger.debug(f"Car {car_idx} at x={car_x_mid:.1f} did not match any parking spot")

        free_spots = [spot['id'] for spot in parking_spots if spot['id'] not in occupied_spots]

        return matches, free_spots

    def detect_double_parking(self, car_boxes, matches, parking_spots, leak_threshold=0.1):
        """
        Detect double parking violations.
        
        A double parking violation occurs when a car assigned to one spot
        significantly overlaps with a neighboring spot.
        
        Args:
            car_boxes (List): List of vehicle bounding boxes
            matches (List[Dict]): List of car-to-spot matches
            parking_spots (List[Dict]): List of parking spots
            leak_threshold (float, optional): Minimum overlap ratio to consider
                as double parking. Defaults to 0.1.
            
        Returns:
            List[Dict]: List of double parking violations
        """
        logger.info("\nChecking for double parking")
        double_parking = []
        spot_id_to_spot = {spot['id']: spot for spot in parking_spots}

        for match in matches:
            car_idx = match['car_idx']
            car_box_coords = match['car_box']
            assigned_spot_id = match['matched_spot_id']
            car_poly = box(*car_box_coords)
            car_area = car_poly.area

            neighbor_ids = [assigned_spot_id - 2, assigned_spot_id + 2]

            for neighbor_id in neighbor_ids:
                if neighbor_id not in spot_id_to_spot:
                    continue

                neighbor_spot = spot_id_to_spot[neighbor_id]
                intersection = car_poly.intersection(neighbor_spot['polygon'])

                if not intersection.is_empty:
                    leak_area = intersection.area
                    leak_ratio = leak_area / car_area

                    logger.debug(f"Car {car_idx} (p{assigned_spot_id})  p{neighbor_id} overlap: {leak_ratio * 100:.4f}%")
                    logger.debug(f"     Exact ratio: {leak_ratio}")
                    logger.debug(f"     Threshold: {leak_threshold}")
                    logger.debug(f"     Comparison result: {leak_ratio >= leak_threshold}")

                    epsilon = 1e-10
                    if leak_ratio >= (leak_threshold - epsilon):
                        double_parking.append({
                            'car_idx': car_idx,
                            'car_box': car_box_coords,
                            'spot_id': assigned_spot_id,
                            'neighbor_id': neighbor_id,
                            'overlap_ratio': float(leak_ratio)
                        })
                        logger.info(f"    DOUBLE PARKING DETECTED")
                    else:
                        logger.debug(f"    Below threshold")

        return double_parking

    def detect_illegal_cars(self,
                            car_boxes: list,
                            parking_spots: list,
                            cam_num: int,
                            buffer_px: int = 10) -> list:
        """
        Detect vehicles parked in illegal areas.
        
        Uses camera-specific logic to determine if a car is parked in a non-parking area.
        
        Args:
            car_boxes (list): List of vehicle bounding boxes
            parking_spots (list): List of parking spots
            cam_num (int): Camera ID
            buffer_px (int, optional): Buffer around parking spots in pixels. Defaults to 10.
            
        Returns:
            list: List of illegally parked vehicles
        """
        illegal_cars = []

        # Build polygons for every parking spot
        spot_polys = [Polygon(s["coords"]).buffer(buffer_px) for s in parking_spots]

        def inside_any_bay(pt: Point) -> bool:
            return any(poly.contains(pt) for poly in spot_polys)

        def pick(coords, k):
            return max(coords, key=k), min(coords, key=k)

        if cam_num == 1:
            p1 = next(s for s in parking_spots if s["id"] == 1)
            p2 = next(s for s in parking_spots if s["id"] == 2)
            rt1, _ = pick(p1["coords"], lambda c: (c[0], -c[1]))
            rb2, _ = pick(p2["coords"], lambda c: (c[0], c[1]))
            line_start = rt1
            line_end = rb2
            offs = lambda x, y: self.is_point_right_of_line((x, y),
                                                            line_start,
                                                            line_end)

        elif cam_num == 2:
            p1 = next(s for s in parking_spots if s["id"] == 1)
            p2 = next(s for s in parking_spots if s["id"] == 2)
            _, lt2 = pick(p2["coords"], lambda c: (c[0], -c[1]))
            _, lb1 = pick(p1["coords"], lambda c: (c[0], c[1]))
            line_start = lt2
            line_end = lb1
            offs = lambda x, y: self.is_point_left_of_line((x, y),
                                                           line_start,
                                                           line_end)
        else:
            y_thresh = max(max(pt[1] for pt in s["coords"]) for s in parking_spots) + 20
            offs = lambda _x, y: y > y_thresh

        for i, (x1, y1, x2, y2) in enumerate(car_boxes):
            cx, cy = self.get_car_mid_bottom([x1, y1, x2, y2])
            p = Point(cx, cy)

            if inside_any_bay(p):
                continue
            if offs(cx, cy):
                illegal_cars.append({
                    "car_idx": i,
                    "car_box": [x1, y1, x2, y2],
                    "reason": "Parked in prohibited area"
                })

        return illegal_cars

    def analyze_parking(self,
                        image,
                        car_boxes,
                        parking_data_path,
                        camera_id):
        """
        Analyze parking spots and detect violations.
        
        This method performs a complete parking analysis:
        1. Loads parking spot data
        2. Detects illegally parked vehicles
        3. Matches cars to parking spots
        4. Detects double parking violations
        
        Args:
            image (numpy.ndarray): Input image
            car_boxes (List): List of vehicle bounding boxes
            parking_data_path (str): Path to parking spot data JSON file
            camera_id (int): Camera ID
            
        Returns:
            Tuple[numpy.ndarray, Dict]: A tuple containing:
                - Annotated image
                - Parking analysis results
        """
        spots = self.load_parking_spots(parking_data_path, camera_id)

        illegal_cars = self.detect_illegal_cars(car_boxes, spots, camera_id)
        illegal_idx = {c["car_idx"] for c in illegal_cars}

        candidate_boxes = [box for j, box in enumerate(car_boxes) if j not in illegal_idx]
        candidate_to_idx = {k: j for k, j in enumerate(i for i in range(len(car_boxes))
                                                       if i not in illegal_idx)}

        matches_local, free_spots = self.match_cars_to_parking_spots(
            candidate_boxes,
            spots,
            column_threshold=60 if camera_id == 1 else 30
        )

        matches, matched_idx = [], set()
        for m in matches_local:
            original = candidate_to_idx[m["car_idx"]]
            matches.append({
                "car_idx": original,
                "car_box": m["car_box"],
                "matched_spot_id": m["matched_spot_id"]
            })
            matched_idx.add(original)

        double_parking = self.detect_double_parking(car_boxes, matches, spots)
        dp_idx = {d["car_idx"] for d in double_parking}

        illegal_cars = [c for c in illegal_cars
                        if c["car_idx"] not in matched_idx
                        and c["car_idx"] not in dp_idx]

        results = {
            "matches": matches,
            "double_parking": double_parking,
            "illegal_cars": illegal_cars,
            "free_spots": free_spots,
            "occupied_spots": [s["id"] for s in spots if s["id"] not in free_spots],
            "total_spots": len(spots)
        }

        return image, results



    def _save_new_violations(self, violations: List[Dict[str, Any]], violation_type_field: str = "reason"):
        """
        Save detected violations to Firestore database.
        
        Args:
            violations (List[Dict[str, Any]]): List of violation data
            violation_type_field (str, optional): Field name in the violation data
                that contains the violation type. Defaults to "reason".
                
        Raises:
            Exception: If there's an error saving to Firestore
        """
        if not self.firestore_enabled:
            return

        batch = self.firestore_manager.db.batch()
        now_ts = firestore.SERVER_TIMESTAMP

        for vio in violations:
            lp_info = vio.get("license_plate") or {}
            plate = lp_info.get("plate") or lp_info.get("text")
            if not plate:
                continue

            plate = plate.upper().strip()
            if self.firestore_manager.violation_exists(plate):
                continue

            if "spot_id" in vio and "neighbor_id" in vio:
                violation_type = "DOUBLE PARK"
            else:
                violation_type = vio.get(violation_type_field, "UNKNOWN")

            doc_ref = self.firestore_manager.db.collection('violation').document()
            batch.set(doc_ref, {
                "violationType": violation_type,
                "violationTime": now_ts,
                "vehicleID": vio.get("vehicleID", ""),
                "licensePlate": plate,
                "show": False,
            })

        try:
            batch.commit()
            logger.info("New violations saved")
        except Exception as e:
            logger.error(f"Error saving new violations: {e}")




from flask import Flask, request, jsonify, send_from_directory
import os

@app.route('/detect', methods=['POST'])
def detect_vehicles():
    """
    Flask route to handle vehicle detection requests.
    
    Accepts an uploaded image file, processes it for vehicle detection and
    parking analysis, and returns the results as JSON.
    
    Request Parameters:
        - image: Image file to process
        - parking_data_path (optional): Path to parking spot data JSON file
        - regions (optional): Comma-separated list of license plate region codes
        
    Returns:
        JSON response with detection results or error message
    """
    if 'image' not in request.files:
        return jsonify({'error': 'No image file provided'}), 400

    image_file = request.files['image']

    temp_image_path = 'temp_upload.jpg'
    image_file.save(temp_image_path)

    parking_data_path = request.form.get('parking_data_path', None)
    regions = request.form.get('regions', 'jo').split(',')

    try:
        results = detector.process_image(
            image_path=temp_image_path,
            parking_data_path=parking_data_path,
            regions=regions
        )

        return jsonify(results)
    except Exception as e:
        return jsonify({'error': str(e)}), 500
    finally:
        if os.path.exists(temp_image_path):
            os.remove(temp_image_path)


@app.route('/spots', methods=['GET'])
def get_parking_spots():
    """
    Flask route to retrieve parking spot data.
    
    Returns parking spot information for a specific camera.
    
    Query Parameters:
        - parking_data_path: Path to parking spot data JSON file
        - camera_id: Camera ID to filter spots
        
    Returns:
        JSON response with parking spot data or error message
    """
    parking_data_path = request.args.get('parking_data_path', None)
    camera_id = request.args.get('camera_id', None)

    if not parking_data_path or not camera_id:
        return jsonify({'error': 'Missing required parameters: parking_data_path and camera_id'}), 400

    try:
        camera_id = int(camera_id)
        spots = detector.load_parking_spots(parking_data_path, camera_id)

        formatted_spots = []
        for spot in spots:
            formatted_spots.append({
                'id': spot['id'],
                'name': spot['name'],
                'x_center': float(spot['x_center']),
                'bottom_y': float(spot['bottom_y']),
                'coordinates': spot['coords'].tolist()
            })

        return jsonify({'spots': formatted_spots})
    except Exception as e:
        return jsonify({'error': str(e)}), 500
