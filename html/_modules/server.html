"""
Flask server for vehicle detection and parking analysis system.

This module provides a web service that processes parking area images from two camera views,
detects vehicles, analyzes parking spots, and identifies various parking violations.
It exposes several API endpoints for retrieving detection results and parking information.

The server runs a background thread that continuously processes image pairs from two
folders at regular intervals, making the latest results available through the API.
"""

import os
import time
import json
import threading
from typing import Dict, List, Tuple

import cv2
import numpy as np
import torch
from flask import (
    Flask, request, jsonify, send_from_directory
)
from flask_cors import CORS

from merge_results import TwoViewAnalyzer
from detection_engine import VehicleDetector

import logging

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
    handlers=[
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

# Folder paths for the camera images
VIEW1_FOLDER: str = r'..\testingSingleImages\images1'
VIEW2_FOLDER: str = r'..\testingSingleImages\images2'

# Path to the parking spot labels file
PARKING_DATA_PATH: str = 'oldLotsCoordinates.json'

# Region code for license plate recognition
REGIONS: List[str] = ['jo']  # Jordan region code

# Time in seconds between processing iterations
PROCESSING_INTERVAL: int = 60

# Allowed image file extensions
ALLOWED_EXT: Tuple[str, ...] = ('.jpg', '.jpeg', '.png')

app = Flask(__name__)
CORS(app)

try:
    pair_analyzer = TwoViewAnalyzer()
    single_detector = VehicleDetector()
    logger.info("Vehicle detection system was initialized successfully")
except Exception as e:
    logger.warning(f"Error initializing vehicle detection: {e}\nThe system will continue but may have limited functionality")
    pair_analyzer = None
    single_detector = None

# Dictionary to store the latest processed results
latest_results: Dict[int, dict] = {}
latest_results_lock = threading.Lock()

# List of image file pairs from the two camera views
image_pairs: List[Tuple[str, str]] = []
pair_index: int = 0
processing_thread_running: bool = False

def make_serializable(obj):
    """
    Convert complex objects to JSON-serializable types.
    
    Handles PyTorch tensors, NumPy arrays, and various numeric types,
    converting them to standard Python types that can be serialized to JSON.
    
    Args:
        obj: Object to convert
        
    Returns:
        JSON-serializable version of the input object
    """
    if isinstance(obj, torch.Tensor):
        return obj.cpu().detach().numpy().tolist()
    if isinstance(obj, np.ndarray):
        return obj.tolist()
    if isinstance(obj, (np.float32, np.float64)):
        return float(obj)
    if isinstance(obj, (np.int32, np.int64)):
        return float(obj)
    if isinstance(obj, dict):
        return {k: make_serializable(v) for k, v in obj.items()}
    if isinstance(obj, list):
        if len(obj) == 4 and all(isinstance(x, (int, float)) for x in obj):
            return [float(x) for x in obj]
        return [make_serializable(i) for i in obj]
    return obj



def _scan_for_pairs():
    """
    Scan the two camera folders for matching image pairs.
    
    Looks for image files with the same base name in both camera view 
    folders and builds a list of file path pairs. Creates the view folders
    if they don't exist.
    
    Updates the global `image_pairs` list with the found pairs.
    
    Raises:
        Exception: If there's an error scanning the folders
    """
    global image_pairs

    try:
        os.makedirs(VIEW1_FOLDER, exist_ok=True)
        os.makedirs(VIEW2_FOLDER, exist_ok=True)

        f1_files = sorted(f for f in os.listdir(VIEW1_FOLDER) if f.lower().endswith(ALLOWED_EXT))
        f2_files = sorted(f for f in os.listdir(VIEW2_FOLDER) if f.lower().endswith(ALLOWED_EXT))

        f1 = {os.path.splitext(f)[0]: os.path.join(VIEW1_FOLDER, f) for f in f1_files}
        f2 = {os.path.splitext(f)[0]: os.path.join(VIEW2_FOLDER, f) for f in f2_files}

        common = sorted(set(f1) & set(f2))
        image_pairs = [(f1[k], f2[k]) for k in common]
        logger.info(f"[SCAN] {len(image_pairs)} matching pairs found, sorted by name.")
    except Exception as e:
        logger.error(f"[SCAN] Error scanning for pairs: {e}")

def _process_pairs_thread():
    """
    Background thread function for continuous image pair processing.
    
    Runs in an infinite loop as long as `processing_thread_running` is True.
    In each iteration:
    1. Scans for image pairs if none are available
    2. Processes the next image pair in the list
    3. Updates the global `latest_results` with the processing results
    4. Waits for the specified interval before the next iteration
    
    Updates the global variables:
    - pair_index: Current position in the image_pairs list
    - latest_results: Dictionary containing the most recent detection results
    """
    global pair_index
    logger.info("[THREAD] Two‑view processing thread launched.")

    while processing_thread_running:
        if not image_pairs:
            _scan_for_pairs()
            if not image_pairs:
                logger.debug("[THREAD] No pairs yet → sleeping for 10 seconds.")
                time.sleep(10)
                continue

        if pair_index >= len(image_pairs):
            pair_index = 0

        v1, v2 = image_pairs[pair_index]
        logger.info(f"[THREAD] {pair_index + 1}/{len(image_pairs)}  {os.path.basename(v1)}  +  {os.path.basename(v2)}")

        try:
            if pair_analyzer is None:
                logger.warning("[THREAD] Error: pair_analyzer not initialized")
                pair_index += 1
                time.sleep(PROCESSING_INTERVAL)
                continue

            res = pair_analyzer.process_pair(
                v1, v2,
                parking_data_path=PARKING_DATA_PATH,
                regions=tuple(REGIONS)
            )
            with latest_results_lock:
                latest_results[0] = make_serializable(res)
            logger.info("[THREAD] Successfully processed pair")
        except Exception as exc:
            logger.error(f"[THREAD] ERROR while processing pair: {exc}")
            import traceback
            traceback.print_exc()

        pair_index += 1
        time.sleep(PROCESSING_INTERVAL)


def _start_processing_thread():
    """
    Start the background thread for processing image pairs.
    
    Creates and starts a daemon thread that runs the `_process_pairs_thread`
    function if it's not already running and the pair analyzer is available.
    
    Updates the global variable:
    - processing_thread_running: Set to True if the thread is started
    """
    global processing_thread_running
    if not processing_thread_running and pair_analyzer is not None:
        processing_thread_running = True
        threading.Thread(
            target=_process_pairs_thread,
            daemon=True
        ).start()
        logger.info("[THREAD] Background processing thread started")
    else:
        logger.warning("[THREAD] Warning: Cannot start processing thread (pair_analyzer not available)")


@app.route('/latest_results', methods=['GET'])
def api_latest_results():
    """
    API endpoint that returns the latest detection results.
    
    Returns:
        Flask response: JSON containing the latest processing results,
        or an error message if no results are available yet.
        
    HTTP Status Codes:
        200: Success
        404: No results available yet
    """
    with latest_results_lock:
        if latest_results:
            return jsonify(next(iter(latest_results.values())))
    return jsonify({"error": "No results yet"}), 404


@app.route('/status', methods=['GET'])
def api_status():
    """
    API endpoint that returns the current status of the detection system.
    
    Returns:
        Flask response: JSON containing status information including:
        - Whether the processing thread is running
        - Number of image pairs in cache
        - Current pair index being processed
        - Whether results are available
        - Firestore database connection status
        - Availability of the detector and pair analyzer
        
    HTTP Status Codes:
        200: Success
    """
    firestore_status = "Not available"
    if single_detector and hasattr(single_detector, 'firestore_enabled'):
        firestore_status = "Enabled" if single_detector.firestore_enabled else "Disabled"

    return jsonify({
        "running": processing_thread_running,
        "pairs_in_cache": len(image_pairs),
        "current_pair_index": pair_index,
        "has_result": bool(latest_results),
        "firestore_status": firestore_status,
        "detector_available": single_detector is not None,
        "pair_analyzer_available": pair_analyzer is not None
    })


@app.route('/detect', methods=['POST'])
def api_detect_single():
    """
    API endpoint for processing a single uploaded image.
    
    Accepts an image file upload, processes it using the single detector,
    and returns the detection results.
    
    Required form data:
        - image: Image file to process
        
    Optional form data:
        - parking_data_path: Path to parking spot data JSON file
        - regions: Comma-separated list of license plate region codes
        
    Returns:
        Flask response: JSON containing the detection results or error message
        
    HTTP Status Codes:
        200: Success
        400: No image file provided
        500: Error during processing
        503: Vehicle detector not available
    """
    if single_detector is None:
        return jsonify({"error": "Vehicle detector not available"}), 503

    if 'image' not in request.files:
        return jsonify({"error": "No image file provided"}), 400

    tmp_img = 'temp_upload.jpg'
    request.files['image'].save(tmp_img)

    try:
        res = single_detector.process_image(
            image_path=tmp_img,
            parking_data_path=request.form.get('parking_data_path', PARKING_DATA_PATH),
            regions=request.form.get('regions', ','.join(REGIONS)).split(',')
        )
        return jsonify(make_serializable(res))
    except Exception as exc:
        return jsonify({"error": str(exc)}), 500
    finally:
        if os.path.exists(tmp_img):
            os.remove(tmp_img)


@app.route('/spots', methods=['GET'])
def api_spots():
    """
    API endpoint that returns parking spot information.
    
    Retrieves and formats parking spot data for a specific camera view.
    
    Query parameters:
        camera_id (int, optional): Camera ID (default: 1)
        
    Returns:
        Flask response: JSON containing the formatted parking spot data
        
    HTTP Status Codes:
        200: Success
        500: Error retrieving or formatting spot data
        503: Vehicle detector not available
    """
    if single_detector is None:
        return jsonify({"error": "Vehicle detector not available"}), 503

    camera_id = request.args.get('camera_id', default=1, type=int)
    try:
        spots = single_detector.load_parking_spots(PARKING_DATA_PATH, camera_id)
        formatted = [{
            "id": s['id'],
            "name": s.get('name', f"P{s['id']}"),
            "x_center": float(s['x_center']),
            "bottom_y": float(s['bottom_y']),
            "coordinates": np.asarray(s['coords']).tolist()
        } for s in spots]
        return jsonify({"spots": formatted})
    except Exception as exc:
        return jsonify({"error": str(exc)}), 500
